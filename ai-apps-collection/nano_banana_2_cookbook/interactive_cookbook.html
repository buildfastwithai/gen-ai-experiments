01

Generate Images

Text to image. Returns both text and image parts.

🖼️Output appears here
Python
from google import genai
from google.genai import types

client = genai.Client(api_key="YOUR_API_KEY")

response = client.models.generate_content(
    model="gemini-3.1-flash-image-preview",
    contents="A golden retriever on a beach...",
    config=types.GenerateContentConfig(
        response_modalities=['Text', 'Image']
    )
)

for part in response.parts:
    if img := part.as_image():
        img.save("out.png")

05

Chat Mode

Multi-turn conversation. Model remembers context and maintains character consistency.

💬Chat history appears here
Python
chat = client.chats.create(
    model="gemini-3.1-flash-image-preview"
)

# First turn
response = chat.send_message("Create a toy fox figurine...")

# Follow-up - model remembers the fox
response = chat.send_message("Add a helmet with a blue planet")

# Keep iterating
response = chat.send_message("Move it to a beach")
06

Thinking

Model reasons before generating. Use for complex prompts. Levels: Minimal, High.

🧠Thoughts + image appear here
Python
response = client.models.generate_content(
    model="gemini-3.1-flash-image-preview",
    contents="Create an unusual viral image...",
    config=types.GenerateContentConfig(
        response_modalities=['Text', 'Image'],
        thinking_config=types.ThinkingConfig(
            thinking_level="High",  # "Minimal" | "High"
            include_thoughts=True
        )
    )
)

# Access thoughts
for part in response.parts:
    if part.thought:
        print(part.text)

08

Image Grounding

Search for reference images on Google. NB2 exclusive. Perfect for landmarks, species, specific visuals.

🖼️Grounded image appears here
Python
response = client.models.generate_content(
    model="gemini-3.1-flash-image-preview",
    contents="A Timareta Thelxione butterfly...",
    config=types.GenerateContentConfig(
        response_modalities=["IMAGE"],
        tools=[
            types.Tool(google_search=types.GoogleSearch(
                search_types=types.SearchTypes(
                    web_search=types.WebSearch(),
                    image_search=types.ImageSearch()  # NB2 only
                )
            ))
        ]
    )
)
